void pciSetup(byte pin)
{
    *digitalPinToPCMSK(pin) |= bit (digitalPinToPCMSKbit(pin));  // enable pin
    PCIFR  |= bit (digitalPinToPCICRbit(pin)); // clear any outstanding interrupt
    PCICR  |= bit (digitalPinToPCICRbit(pin)); // enable interrupt for the group
}



ISR (PCINT2_vect)//left // handle pin change interrupt for D0 to D7 here
{
    static unsigned long last_interrupt_time = 0;
 	unsigned long interrupt_time = millis();
 	// If interrupts come faster than 200ms, assume it's a bounce and ignore
 	if (interrupt_time - last_interrupt_time > 75) 
 	{
      	
        
   		if(digitalRead(6) == HIGH && leftPaddle == 0 && X_left_paddle == 0)
        {
          //Do nothing;
        }
        else if(digitalRead(6) == HIGH)
        {
          lcd.setCursor(0,X_left_paddle);
          lcd.print(" ");
          leftPaddle = (leftPaddle + 1)%2;
          if(leftPaddle == 1)
          {  
          	X_left_paddle = (X_left_paddle + 1)%2;
          } 
          lcd.createChar(5,pLeft[leftPaddle]);
          lcd.setCursor(0,X_left_paddle);
          lcd.write((byte)5);
          
          lcd.createChar(1,array[dy][dx]);  // Value is a pointer to an array of 8 bytes
  		  lcd.setCursor(Y,X);
  		  lcd.write(1);
        }
        else if(digitalRead(7) == HIGH && leftPaddle == 1 && X_left_paddle == 1)
        {
          //Do nothing;
        }
        else if(digitalRead(7) == HIGH)
        {
          lcd.setCursor(0,X_left_paddle);
          lcd.print(" ");
          leftPaddle = (leftPaddle + 1)%2;
          if(leftPaddle == 0)
          {  
          	X_left_paddle = (X_left_paddle + 1)%2;
          } 
          lcd.createChar(5,pLeft[leftPaddle]);
          lcd.setCursor(0,X_left_paddle);
          lcd.write((byte)5);
          
          lcd.createChar(1,array[dy][dx]);  // Value is a pointer to an array of 8 bytes
  		  lcd.setCursor(Y,X);
  		  lcd.write(1);
        }  
 	}
 	last_interrupt_time = interrupt_time;
    delay(2);
 }

void setup()
{
  pinMode(6,INPUT);
  pinMode(7,INPUT);
  pinMode(8,INPUT);
  pinMode(9,INPUT);
  
  lcd.begin(16,2);
  Serial.begin(9600);
  lcd.clear();
  
  int i;
  for(i=6;i<=9;i++)
  {  
    digitalWrite(i,HIGH);
  }     
  
  sei(); 
  
  pciSetup(8);
  pciSetup(9);
  pciSetup(7);
  pciSetup(6);
  
  gameStart();
  
}  

void gameStart()
{
  leftPaddle = 1;
  X_left_paddle =1;
  
  rightPaddle = 1;
  X_right_paddle = 1;
  
  lcd.createChar(2,pLeft[leftPaddle]);
  lcd.setCursor(0,X_left_paddle);
  lcd.write((byte)2);
  
  lcd.createChar(3,pRight[rightPaddle]);
  lcd.setCursor(10,X_right_paddle);
  lcd.write((byte)3);
  
  X = 0;
  Y = 9;
  
  dx = 4;
  dy = 0;
  vert = 1;
  hori = -1;
} 

void loop()
{
  ballTravel();
  delay(4);
  if(hori == -1 && Y == 1 && dx == 0)
  {
    leftPaddleCheck();
  }
  else if(hori == 1 && Y == 9 && dx == 4)
  {
    rightPaddleCheck();
  }  
  else
  {
  	ballPosUpdation();
  }
}

void ballPosUpdation()
{
  if(vert == 1 && hori == 1)
  {
     if(dy > 0)
     {
       dy = dy-1;
       dx = (dx+1)%5;
       if(dx == 0)
       {
         Y = Y+1;
       }  
     }
     else if(dy == 0 && X == 0)
     {
       dx = (dx+1)%5;
       if(dx == 0)
       {
         Y = Y+1;
       }  
       dy = dy+1;
       hori = 1;
       vert = -1;
     }
     else if(dy == 0 && X == 1)
     {
       dy = 7;
       dx = (dx+1)%5;
       if(dx == 0)
       {
         Y = Y+1;
       }  
       X = 0;
     }    
  }
  else if(vert == -1 && hori == 1)
  {
     if(dy == 7 && X == 1)//bounce back from bottom
     {
       dx = (dx+1)%5;
       dy = dy-1;
       hori = 1;
       vert = 1;
       if(dx == 0)
       {
         Y = Y+1;
       }  
     }
     else //Normal in-cell
     {
       dy = (dy+1)%8;
       dx = (dx+1)%5;
       if(dx == 0)
       {
         Y = Y+1;
       }
       if(dy == 0)
       {
         X = 1;
       }  
     }
  }
  else if(vert == 1 && hori == -1)
  {
     if(dy == 0 && X == 0)
     {
       dx = dx-1;
       dy = dy+1;
       hori = -1;
       vert = -1;
       if(dx == -1)
       {
         dx = 4;
         Y = Y-1;
       }  
     }
     else//in-cell transition
     {
       dy = dy-1;
       dx = dx-1;
       if(dy == -1)
       {
         dy = 7;
         X = 0;
       }
       if(dx == -1)
       {
         dx = 4;
         Y = Y-1;
       }  
     }
  }
  else if(vert == -1 && hori == -1)
  {
     if(dy == 7 && X == 1)//bounce back from bottom
     {
       dx = dx-1;
       dy = dy-1;
       hori = -1;
       vert = 1;
       if(dx == -1)
       {
         dx = 4;
         Y = Y-1;
       }  
     }
     else //Normal in-cell
     {
       dy = (dy+1)%8;
       dx = dx-1;
       if(dy == 0)
       {
         X = 1;
       }
       if(dx == -1)
       {
         dx = 4;
         Y = Y - 1;
       }
     } 
  }
}

void rightPaddleCheck()
{
  if(vert == 1 && hori == 1)
  {
     if(rightPaddle == 0 && X == 0 && 0<=dy<=3)
     {  
       if(dy == 0)
       {
         dy = dy+1;
         vert = -1;
         hori = -1;
       }
       else 
       {
         dy = dy-1;
         vert = 1;
         hori = -1;
       }  
     }
     else if(rightPaddle == 1 && X == 0 && 4<=dy<=7)
     {  
        dy = dy-1;
        vert = 1;
        hori = -1; 
     }
     else if(rightPaddle == 2 && X == 1 && 0<=dy<=3)
     {  
        dy = dy-1;
        vert = 1;
        hori = -1; 
        if(dy == -1)
        {
          dy = 7;
          X = 0;
        }  
     }
     else if(rightPaddle == 3 && X == 1 && 4<=dy<=7)
     {  
        dy = dy-1;
        vert = 1;
        hori = -1;   
     }
     else 
     {
       scoreRight = scoreRight + 1;
     }   
  }
  else if(vert == -1 && hori == 1)
  {
    if(rightPaddle == 0 && X ==0 && 0<=dy<=3)
    {
      dy = dy+1;
      vert = -1;
      hori = -1;
    }  
    else if(rightPaddle == 1 && X ==0 && 4<=dy<=7)
    {
      dy = (dy+1)%8;
      vert = -1;
      hori = -1;
      X = 1;
    }
    else if(rightPaddle == 0 && X == 1 && 0<=dy<=3)
    {
      dy = dy+1;
      vert = -1;
      hori = -1;
    }
    else if(rightPaddle == 1 && X ==1 && 4<=dy<=7)
    {
      dy = dy+1;
      vert = -1;
      hori = -1;
      if(dy == 8)
      {
        dy = dy-1;
        vert = 1;
        hori = -1;
      }
    }
    else
    {
      scoreLeft = scoreLeft + 1;
    }
  }
}
